Select customer_id, MAX(PURCHASE_AMOUNT)
FROM ORDERS
GROUP BY CUSTOMER_ID;

SELECT SALESMAN_ID, MAX(PURCHASE_AMOUNT) AS HIGHEST_AMOUNT
FROM ORDERS
WHERE ORDER_DATE =DATE '2012-08-17'
GROUP BY SALESMAN_ID;

SELECT CUSTOMER_ID, MAX(PURCHASE_AMOUNT) AS HIGHEST_AMOUNT, ORDER_DATE
FROM ORDERS
WHERE PURCHASE_AMOUNT IN (2030,3450,5760,6000)
GROUP BY CUSTOMER_ID,ORDER_DATE;
